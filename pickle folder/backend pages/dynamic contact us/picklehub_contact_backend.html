# models.py
class ContactPage(models.Model):
    hero_title     = CharField(max_length=100)
    hero_subtitle  = CharField(max_length=200)
    biz_name       = CharField(max_length=100)
    address        = TextField()
    map_url        = URLField(blank=True)
    reply_email    = EmailField()

class ContactCard(models.Model):
    icon      = CharField(max_length=10)   # emoji
    title     = CharField(max_length=60)
    value     = CharField(max_length=100)
    sub_text  = CharField(max_length=80)
    url       = CharField(max_length=200)
    bg_color  = CharField(max_length=10)
    is_active = BooleanField(default=True)
    order     = IntegerField(default=0)

class WorkingHour(models.Model):
    day        = CharField(max_length=20)
    open_time  = CharField(max_length=20)
    close_time = CharField(max_length=20)
    is_closed  = BooleanField(default=False)
