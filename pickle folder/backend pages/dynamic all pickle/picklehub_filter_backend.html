📋 How Filters Work with Backend
① Admin edits filter options here
② Saved to FilterConfig Django model
③ Filter sidebar renders from {{ filters }}
④ On Apply → URL params: ?cat=mango&brand=andhra&price=500
⑤ Django view filters Product.objects by params
⑥ Product cards rendered by backend — unchanged
🐍 Django Integration Snippet
# models.py
class FilterCategory(models.Model):
    label = CharField(max_length=80)
    count = IntegerField(default=0)
    is_active = BooleanField(default=True)
    order = IntegerField(default=0)

# views.py — products page
def products(request):
    cats    = FilterCategory.objects.filter(is_active=True)
    brands  = FilterBrand.objects.filter(is_active=True)
    regions = FilterRegion.objects.filter(is_active=True)
    # Apply filters from GET params
    qs = Product.objects.all()
    if request.GET.get('cat'):
        qs = qs.filter(category__slug=request.GET['cat'])
    if request.GET.get('price'):
        qs = qs.filter(price__lte=request.GET['price'])
    context = {'products': qs, 'cats': cats, ...}
